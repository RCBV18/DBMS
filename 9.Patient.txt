-- Doctor table
CREATE TABLE Doctor (
    Doc_ID VARCHAR(10),
    Name VARCHAR(50),
    Specialization VARCHAR(50),
    Allocated_Slot VARCHAR(20)
);

-- Patient table
CREATE TABLE Patient (
    PID VARCHAR(10),
    Name VARCHAR(50),
    Suffering_From VARCHAR(100),
    Bill DECIMAL(10,2)
);

-- Appointment table
CREATE TABLE Appointment (
    App_ID VARCHAR(10),
    PID VARCHAR(10),
    Doc_ID VARCHAR(10),
    Appointment_Date DATE
);
