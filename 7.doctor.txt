-- Doctor table
CREATE TABLE Doctor (
    Doc_Id VARCHAR(10),
    Name VARCHAR(50),
    Allocated_Slot VARCHAR(20),
    Specialization VARCHAR(50)
);

-- Patient table
CREATE TABLE Patient (
    PID VARCHAR(10),
    Name VARCHAR(50),
    suffering_from VARCHAR(100),
    Bill DECIMAL(10,2)
);
